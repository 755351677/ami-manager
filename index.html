    function setSyncStatus(msg, success = true) {

      const el = document.getElementById('syncStatus');

      if (el) { el.textContent = msg; el.className = 'sync-status ' + (success ? 'synced' : ''); }

    }

    function showLoading(show) {

      document.getElementById('loading').style.display = show ? 'block' : 'none';

      document.getElementById('table').style.display = show ? 'none' : 'table';

    }

    cardsRef.once('value').then((snapshot) => {

      if (firebaseData && firebaseData.length > 0) {

        cards = firebaseData;

        setSyncStatus('已同步');

        cards = localData;

        cardsRef.set(cards).then(() => setSyncStatus('已同步'));

      render();

      showLoading(false);

      render();

    });

    cardsRef.onValue((snapshot) => {

      if (!isFirstLoad) {

        const firebaseData = snapshot.val();

        if (firebaseData) {

          cards = firebaseData;

          render();

        }

      }

    });

    function saveToFirebase() {

      cardsRef.set(cards).then(() => setSyncStatus('已同步')).catch(() => setSyncStatus('同步失败', false));

    }

    function toggleSelect(index) {

      const idx = selected.indexOf(index);

      if (idx > -1) selected.splice(idx, 1);

      else selected.push(index);

      render();

    }

    function toggleSelectAll() {

      const checked = document.getElementById('selectAll').checked;

      const search = document.getElementById('searchInput').value.toLowerCase();

      const typeFilter = document.getElementById('typeFilter').value;

      const statusFilter = document.getElementById('statusFilter').value;

      const visibleCards = cards.filter(c => {

        const matchSearch = c.kami.toLowerCase().includes(search) || (c.note && c.note.toLowerCase().includes(search));

        const matchType = !typeFilter || c.type === typeFilter;

        const matchStatus = !statusFilter || c.status === statusFilter;

        return matchSearch && matchType && matchStatus && c.status === '可用';

      });

      if (checked) selected = visibleCards.map((c) => cards.indexOf(c));

      else selected = [];

      render();

    }

    function batchMarkUsed() {

      if (selected.length === 0) { alert('请先选择卡密'); return; }

      if (confirm('确认将 ' + selected.length + ' 条卡密标记为已用？')) {

        selected.forEach(i => { if (cards[i].status === '可用') cards[i].status = '已用'; });

        selected = [];

        saveToFirebase();

        render();

      }

    }

    function copySelected() {

      if (selected.length === 0) { alert('请先选择卡密'); return; }

      const kamis = selected.map(i => cards[i].kami).join('\n');

      navigator.clipboard.writeText(kamis).then(() => alert('已复制 ' + selected.length + ' 条卡密'));

    }

    function openBatchNoteModal() {

      if (selected.length === 0) { alert('请先选择卡密'); return; }

      document.getElementById('batchNoteModal').classList.add('active');

      document.getElementById('batchNoteInput').value = '';

    }

    function closeBatchNoteModal(e) {

      if (e && e.target !== e.currentTarget) return;

      document.getElementById('batchNoteModal').classList.remove('active');

    }

    function applyBatchNote() {

      const note = document.getElementById('batchNoteInput').value.trim();

      if (!note) { alert('请输入备注内容'); return; }

      selected.forEach(i => { cards[i].note = note; });

      saveToFirebase();

      closeBatchNoteModal();

      render();

    }

    function copyAndMark(index) {

      const c = cards[index];

      navigator.clipboard.writeText(c.kami).then(() => {

        if (c.status === '可用') { c.status = '已用'; saveToFirebase(); render(); }

      });

    }

    function render() {

      const search = document.getElementById('searchInput').value.toLowerCase();

      const typeFilter = document.getElementById('typeFilter').value;

      const statusFilter = document.getElementById('statusFilter').value;

      const filtered = cards.filter(c => {

        const matchSearch = c.kami.toLowerCase().includes(search) || (c.note && c.note.toLowerCase().includes(search));

        const matchType = !typeFilter || c.type === typeFilter;

        const matchStatus = !statusFilter || c.status === statusFilter;

        return matchSearch && matchType && matchStatus;

      });

      const tbody = document.getElementById('tbody');

      const empty = document.getElementById('empty');

      const table = document.getElementById('table');

      if (filtered.length === 0) { table.style.display = 'none'; empty.style.display = 'block'; return; }

      table.style.display = 'table'; empty.style.display = 'none';

      tbody.innerHTML = filtered.map((c, i) => {

        const realIndex = cards.indexOf(c);

        const badgeClass = c.type === '时间' ? 'badge-time' : 'badge-count';

        const statusClass = c.status === '可用' ? 'badge-success' : 'badge-default';

        const rowClass = c.status === '已用' ? 'class="row-used"' : '';

        const checked = selected.includes(realIndex) ? 'checked' : '';

        const dis = c.status === '已用' ? 'disabled' : '';

        return `<tr ${rowClass}><td><input type="checkbox" ${checked} ${dis} onclick="toggleSelect(${realIndex})"></td><td>${escapeHtml(c.kami)} <span class="copy-btn" onclick="copyAndMark(${realIndex})">📋</span></td><td><span class="badge ${badgeClass}">${c.type}</span></td><td>${escapeHtml(c.note || '')}</td><td><span class="badge ${statusClass}">${c.status}</span></td><td class="actions"><button class="btn btn-small btn-edit" onclick="edit(${realIndex})">编辑</button><button class="btn btn-small btn-del" onclick="del(${realIndex})">删除</button></td></tr>`;

      }).join('');

    }

    function escapeHtml(str) { const div = document.createElement('div'); div.textContent = str; return div.innerHTML; }

    function openModal(index = -1) {

      editingIndex = index;

      document.getElementById('modalOverlay').classList.add('active');

      document.getElementById('modalTitle').textContent = index === -1 ? '添加卡密' : '编辑卡密';

      if (index === -1) {

        document.getElementById('kamiInput').value = '';

        document.getElementById('typeInput').value = '时间';

        document.getElementById('noteInput').value = '';

        document.getElementById('statusInput').value = '可用';

      } else {

        const c = cards[index];

        document.getElementById('kamiInput').value = c.kami;

        document.getElementById('typeInput').value = c.type;

        document.getElementById('noteInput').value = c.note || '';

        document.getElementById('statusInput').value = c.status;

      }

    }

    function closeModal(e) {

      if (e && e.target !== e.currentTarget) return;

      document.getElementById('modalOverlay').classList.remove('active');

    }

    function edit(index) { openModal(index); }

    function del(index) {

      if (confirm('确定删除这条卡密？')) { cards.splice(index, 1); saveToFirebase(); render(); }

    }

    function save() {

      const kami = document.getElementById('kamiInput').value.trim();

      const type = document.getElementById('typeInput').value;

      const note = document.getElementById('noteInput').value.trim();

      const status = document.getElementById('statusInput').value;

      if (!kami) { alert('请输入卡密'); return; }

      const card = { kami, type, note, status };

      if (editingIndex === -1) cards.unshift(card);

      else cards[editingIndex] = card;

      saveToFirebase();

      closeModal();

      render();

    }
